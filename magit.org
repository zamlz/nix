* Package Install
:PROPERTIES:
:header-args:emacs-lisp: :tangle ~/.config/emacs/lisp/init-magit.el :comments both :mkdirp yes
:END:

#+begin_src emacs-lisp
(use-package magit
  :hook (with-editor-mode . evil-insert-state))
#+end_src
